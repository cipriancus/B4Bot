os.environ["JAVAHOME"] = "C:\Program Files\Java\jre1.8.0_111"
os.environ["STANFORD_PARSER"] = "D:\Stanford\stanford-parser-full-2015-12-09"
os.environ["STANFORD_MODELS"] = "D:\Stanford\stanford-parser-full-2015-12-09"
